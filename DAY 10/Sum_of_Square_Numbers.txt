class Solution {
public:
    bool judgeSquareSum(int c) {
        long long int left = 0;
        long long int right = sqrt(c);  
        
        while (left <= right) {
            long long int squaresum = left * left + right * right;  
            
            if (squaresum == c) {
                return true;  
            }
            if (squaresum < c) {
                left++; 
            } else {
                right--;  
            }
        }
        
        return false; 
    }
};
